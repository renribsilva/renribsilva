```
poesia <- função(ideia, linguagem) {

  se (!ideia) {
    pare("Faça qualquer coisa que o inspire")
  }
  
  forma <- vazio
  verso <- vazio

  para (i em 1:representações(ideia)) {
    forma[i] <- combinar(linguagem[palavras])
  }

  se (!é.vazio(forma)) {
    verso <- como.texto(forma)
  } se não {
    pare("Deixe a ideia repousar e volte a ela mais tarde")
  }

  retornar verso

}

diagramar(poesia)
```
