```
biblioteca(cultura)

poesia <- função(ideia) {

  se (é.nulo(ideia)) {
    pare("Faça qualquer coisa que te inspire")
  }

  linguagem <- cultura[linguagem]
  representações <- cultura::representações(ideias)  
  forma <- como.texto()
  verso <- nulo

  para (i em 1:representações) {
    forma[i] <- combinar(linguagem[palavras])
  }

  se (forma > 0) {
    verso <- como.verso(forma)
  } se não {
    pare("Deixe a ideia repousar e volte para ela mais tarde")
  }

  retornar verso

}

diagramar(poesia("ideia"))
```
