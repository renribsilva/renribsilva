poesia <- função(ideia, linguagem) {

  se (!(ideia)) {
    pare("Faça qualquer coisa que o inspire")
  }
  
  verso <- vazio
  forma <- vazio

  for (i em 1:comprimento(ideia)) {
    verso[i] <- combinar(linguagem[palavras])
  }

  se (verso > 0) {
    forma <- verso
  } se não {
    pare("Deixe a ideia repousar e volte a ela mais tarde")
  }

  retornar forma

}

postar(poesia)
